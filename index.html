Deploy an Apps Script Web App from your Google Sheet to enable registration and data saving.

Steps: Open Sheet → Extensions → Apps Script → Paste code → Deploy as Web App (Execute as: Me, Access: Anyone) → Copy URL

const SS_ID='YOUR_SHEET_ID';
function doGet(e){
  try{
    const d=JSON.parse(e.parameter.data);
    const ss=SpreadsheetApp.openById(SS_ID);
    if(d.action==='register'){
      let s=ss.getSheetByName('Users');
      if(!s){s=ss.insertSheet('Users');s.appendRow(['FirstName','LastName','Email','Password','PAN','Mobile','DOB','AY','Status','RegisteredOn']);}
      s.appendRow(d.row);
    } else if(d.action==='login'){
      const s=ss.getSheetByName('Users');
      if(!s)return out(e,{success:false,error:'Users sheet not found'});
      const rows=s.getDataRange().getValues();const h=rows[0];
      const ei=h.indexOf('Email'),pi=h.indexOf('Password'),si=h.indexOf('Status');
      const user=rows.slice(1).find(r=>r[ei]===d.email&&r[pi]===d.password);
      if(!user)return out(e,{success:false,error:'Invalid email or password'});
      if(user[si]!=='approved')return out(e,{success:false,error:'Account pending admin approval'});
      const obj={};h.forEach((k,i)=>obj[k]=user[i]);
      return out(e,{success:true,user:obj});
    } else if(d.action==='save_tax'){
      let s=ss.getSheetByName('TaxData');
      if(!s){s=ss.insertSheet('TaxData');s.appendRow(['Email','Name','PAN','FY','AY','GrossIncome','Deductions','TaxableIncome','OldTax','NewTax','TaxPayable','Regime','Status','SavedOn','Time']);}
      s.appendRow(d.row);
    }
    return out(e,{success:true});
  }catch(err){return out(e,{success:false,error:err.toString()});}
}
function out(e,d){const cb=e.parameter.callback||'cb';return ContentService.createTextOutput(cb+'('+JSON.stringify(d)+')').setMimeType(ContentService.MimeType.JAVASCRIPT);}
function doPost(e){return doGet(e);}

Which ITR Form Should You File?
ITR-1
SAHAJ
Salaried up to ₹50L, one house, basic interest income. Most common form.
ITR-2
Individuals/HUF
Capital gains, multiple properties, foreign income, director/ESOP, income >₹50L.
ITR-3
Business/Profession
Income from business or profession. Partners in firm. Detailed P&L required.
ITR-4
SUGAM
Presumptive taxation (44AD, 44ADA). Business ₹2Cr or profession ₹50L.
ITR-5
Firms/LLPs
Partnership firms, LLPs, AOPs, BOIs. Not individuals or companies.
ITR-6
Companies
All companies. Mandatory e-filing with DSC.
